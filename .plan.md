### 第三阶段：发布功能方案选择

## 🚀 发布功能方案对比

### 方案A：基于 Git 的本地发布（推荐 ⭐⭐⭐）

**原理**：本地 Web 界面 → 生成 Markdown 文件 → Git 提交 → 自动部署

**技术栈**：
- 前端：Next.js API Routes + 现有 tiptap 编辑器
- 存储：本地文件系统
- 图床：外部图床链接（GitHub、SM.MS、Imgur 等）
- 部署：GitHub Actions / Vercel

**优点**：
- ✅ 零服务器成本
- ✅ 版本控制（Git 历史）
- ✅ 数据完全在本地
- ✅ 部署简单（push 即部署）
- ✅ 内存占用极低（无需数据库）
- ✅ 图片使用外部图床，零存储压力

**缺点**：
- ❌ 需要本地运行开发服务器
- ❌ 不支持多人协作
- ❌ 需要 Git 知识

**实现复杂度**：⭐⭐（低）

**图片链接预览功能**：
- 支持粘贴图片链接后实时预览
- 支持常见图床格式（GitHub、SM.MS、Imgur、七牛云等）
- 自动验证链接有效性
- 支持拖拽排序多张图片

**代码示例**：
```typescript
// components/ImageLinkInput.tsx
"use client";

import { useState, useEffect } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { Image, Link, X, Eye, EyeOff } from 'lucide-react';

interface ImageLinkInputProps {
  value: string[];
  onChange: (urls: string[]) => void;
  maxImages?: number;
}

export default function ImageLinkInput({ 
  value = [], 
  onChange, 
  maxImages = 9 
}: ImageLinkInputProps) {
  const [inputUrl, setInputUrl] = useState('');
  const [previewIndex, setPreviewIndex] = useState<number | null>(null);
  const [isValidating, setIsValidating] = useState(false);
  const [error, setError] = useState('');

  // 验证图片链接是否有效
  const validateImageUrl = async (url: string): Promise<boolean> => {
    try {
      setIsValidating(true);
      const response = await fetch(url, { method: 'HEAD' });
      const contentType = response.headers.get('content-type');
      return contentType?.startsWith('image/') || false;
    } catch {
      return false;
    } finally {
      setIsValidating(false);
    }
  };

  // 添加图片链接
  const handleAdd = async () => {
    if (!inputUrl.trim()) return;
    
    if (value.length >= maxImages) {
      setError(`最多只能添加 ${maxImages} 张图片`);
      return;
    }

    // 检查是否已存在
    if (value.includes(inputUrl.trim())) {
      setError('该链接已存在');
      return;
    }

    // 验证链接
    const isValid = await validateImageUrl(inputUrl.trim());
    if (!isValid) {
      setError('无效的图片链接，请检查 URL 是否正确');
      return;
    }

    onChange([...value, inputUrl.trim()]);
    setInputUrl('');
    setError('');
  };

  // 删除图片
  const handleRemove = (index: number) => {
    const newValue = value.filter((_, i) => i !== index);
    onChange(newValue);
    if (previewIndex === index) {
      setPreviewIndex(null);
    }
  };

  return (
    <div className="space-y-4">
      {/* 输入区域 */}
      <div className="flex gap-2">
        <div className="flex-1 relative">
          <Link className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-slate-400" />
          <input
            type="url"
            value={inputUrl}
            onChange={(e) => {
              setInputUrl(e.target.value);
              setError('');
            }}
            placeholder="粘贴图片链接 (https://...)"
            className="w-full pl-10 pr-4 py-3 bg-white/50 dark:bg-slate-800/50 border border-slate-200 dark:border-slate-700 rounded-xl focus:outline-none focus:ring-2 focus:ring-indigo-500 text-sm"
            onKeyDown={(e) => e.key === 'Enter' && handleAdd()}
          />
        </div>
        <button
          onClick={handleAdd}
          disabled={isValidating || !inputUrl.trim()}
          className="px-6 py-3 bg-indigo-500 text-white rounded-xl font-bold text-sm hover:bg-indigo-600 disabled:opacity-50 disabled:cursor-not-allowed transition-all"
        >
          {isValidating ? '验证中...' : '添加'}
        </button>
      </div>

      {/* 错误提示 */}
      {error && (
        <p className="text-red-500 text-sm">{error}</p>
      )}

      {/* 图片列表 */}
      {value.length > 0 && (
        <div className="grid grid-cols-2 sm:grid-cols-3 gap-3">
          {value.map((url, index) => (
            <div key={url} className="relative group rounded-xl overflow-hidden border border-slate-200 dark:border-slate-700">
              <img
                src={url}
                alt={`图片 ${index + 1}`}
                className="w-full h-32 object-cover"
                onError={(e) => {
                  (e.target as HTMLImageElement).src = '/placeholder-image.png';
                }}
              />
              
              {/* 操作按钮 */}
              <div className="absolute inset-0 bg-black/50 opacity-0 group-hover:opacity-100 transition-opacity flex items-center justify-center gap-2">
                <button
                  onClick={() => setPreviewIndex(previewIndex === index ? null : index)}
                  className="p-2 bg-white/20 rounded-lg hover:bg-white/30 transition-colors"
                >
                  <Eye className="w-4 h-4 text-white" />
                </button>
                <button
                  onClick={() => handleRemove(index)}
                  className="p-2 bg-red-500/80 rounded-lg hover:bg-red-500 transition-colors"
                >
                  <X className="w-4 h-4 text-white" />
                </button>
              </div>

              {/* 序号 */}
              <div className="absolute top-2 left-2 bg-black/50 text-white text-xs px-2 py-1 rounded">
                {index + 1}
              </div>
            </div>
          ))}
        </div>
      )}

      {/* 预览区域 */}
      <AnimatePresence>
        {previewIndex !== null && value[previewIndex] && (
          <motion.div
            initial={{ opacity: 0, height: 0 }}
            animate={{ opacity: 1, height: 'auto' }}
            exit={{ opacity: 0, height: 0 }}
            className="overflow-hidden"
          >
            <div className="relative bg-slate-100 dark:bg-slate-800 rounded-xl p-4">
              <button
                onClick={() => setPreviewIndex(null)}
                className="absolute top-2 right-2 p-1 bg-slate-200 dark:bg-slate-700 rounded-lg hover:bg-slate-300 dark:hover:bg-slate-600"
              >
                <EyeOff className="w-4 h-4" />
              </button>
              <img
                src={value[previewIndex]}
                alt="预览"
                className="max-w-full max-h-96 mx-auto rounded-lg"
              />
              <p className="text-center text-sm text-slate-500 mt-2">
                图片 {previewIndex + 1} / {value.length}
              </p>
            </div>
          </motion.div>
        )}
      </AnimatePresence>

      {/* 提示信息 */}
      <p className="text-xs text-slate-400">
        支持常见图床链接（GitHub、SM.MS、Imgur、七牛云等），最多 {maxImages} 张图片
      </p>
    </div>
  );
}
```

**API Routes 示例**：
```typescript
// app/api/posts/route.ts
import fs from 'fs';
import path from 'path';

export async function POST(request: Request) {
  const { title, content, tags, series, images } = await request.json();
  
  const slug = title.toLowerCase().replace(/\s+/g, '-');
  const frontmatter = `---
title: "${title}"
date: "${new Date().toISOString()}"
tags: ${JSON.stringify(tags)}
series: ${JSON.stringify(series)}
images: ${JSON.stringify(images)}
---

${content}`;
  
  fs.writeFileSync(
    path.join(process.cwd(), 'posts', `${slug}.md`),
    frontmatter
  );
  
  return Response.json({ success: true, slug });
}
```

### 代码块复制按钮实现

```typescript
// components/CodeBlock.tsx
"use client";

import { useState } from 'react';

export default function CodeBlock({ children, language }: { children: string; language?: string }) {
  const [copied, setCopied] = useState(false);

  const handleCopy = async () => {
    await navigator.clipboard.writeText(children);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  return (
    <div className="relative group">
      {language && (
        <div className="absolute top-2 right-2 text-xs text-slate-400 bg-slate-700 px-2 py-1 rounded">
          {language}
        </div>
      )}
      <button
        onClick={handleCopy}
        className="absolute top-2 right-2 opacity-0 group-hover:opacity-100 transition-opacity bg-slate-700 hover:bg-slate-600 text-white px-2 py-1 rounded text-xs"
      >
        {copied ? '✓ 已复制' : '复制'}
      </button>
      <pre><code className={`language-${language}`}>{children}</code></pre>
    </div>
  );
}
```

### 系列文章导航实现

```typescript
// components/SeriesNav.tsx
"use client";

import Link from 'next/link';

interface SeriesPost {
  slug: string;
  title: string;
  order: number;
}

export default function SeriesNav({ 
  currentSlug, 
  series 
}: { 
  currentSlug: string; 
  series: { name: string; posts: SeriesPost[] } 
}) {
  const currentIndex = series.posts.findIndex(p => p.slug === currentSlug);
  const prevPost = currentIndex > 0 ? series.posts[currentIndex - 1] : null;
  const nextPost = currentIndex < series.posts.length - 1 ? series.posts[currentIndex + 1] : null;

  return (
    <div className="bg-white/60 dark:bg-slate-800/50 rounded-3xl p-6 border border-white/40 dark:border-white/10">
      <h3 className="font-bold text-slate-900 dark:text-white mb-4">
        📚 {series.name}
      </h3>
      
      <div className="flex justify-between items-center">
        {prevPost ? (
          <Link href={`/posts/${prevPost.slug}`} className="text-indigo-500 hover:text-indigo-600">
            ← {prevPost.title}
          </Link>
        ) : <div />}
        
        <span className="text-sm text-slate-500">
          {currentIndex + 1} / {series.posts.length}
        </span>
        
        {nextPost ? (
          <Link href={`/posts/${nextPost.slug}`} className="text-indigo-500 hover:text-indigo-600">
            {nextPost.title} →
          </Link>
        ) : <div />}
      </div>
    </div>
  );
}
```
